  Header unset ETag
  <FilesMatch "\.(ico|jpe?g|png|gif|swf|css|gz|woff|woff2)$">
    Header set Cache-Control "max-age=31536000, public"
  </FilesMatch>
  <FilesMatch "\.(js)$">
    Header set Cache-Control "max-age=31536000, public"
  </FilesMatch>
</IfModule>
